> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Create a shared demo org with a TMDB-powered knowledge base that new users auto-join as viewers.

**Architecture:** Admin seed endpoint (`POST /api/v1/admin/seed-demo`) fetches movies from TMDB, converts to markdown documents, stores via SeaweedFS, and enqueues chunking+embedding jobs on the existing Asynq pipeline. New users auto-join the demo org on first login. The endpoint is idempotent and auth-gated with session OR `X-Seed-Key` header.

**Tech Stack:** Go, Gin, TMDB REST API, Asynq, pgx, SeaweedFS, SuperTokens

package tmdb

// Genre represents a TMDB genre.
type Genre struct {
	ID   int    `json:"id"`
	Name string `json:"name"`
}

// GenreListResponse is the response from GET /genre/movie/list.
type GenreListResponse struct {
	Genres []Genre `json:"genres"`
}

// MovieSummary is a movie entry from the discover endpoint.
type MovieSummary struct {
	ID          int     `json:"id"`
	Title       string  `json:"title"`
	Overview    string  `json:"overview"`
	ReleaseDate string  `json:"release_date"`
	VoteAverage float64 `json:"vote_average"`
	VoteCount   int     `json:"vote_count"`
}

// DiscoverResponse is the response from GET /discover/movie.
type DiscoverResponse struct {
	Results []MovieSummary `json:"results"`
}

// MovieDetail is the full movie detail with appended credits, reviews, keywords.
type MovieDetail struct {
	ID          int     `json:"id"`
	Title       string  `json:"title"`
	Overview    string  `json:"overview"`
	ReleaseDate string  `json:"release_date"`
	VoteAverage float64 `json:"vote_average"`
	VoteCount   int     `json:"vote_count"`
	Runtime     int     `json:"runtime"`
	Genres      []Genre `json:"genres"`
	Credits     Credits `json:"credits"`
	Reviews     Reviews `json:"reviews"`
	Keywords    Keywords `json:"keywords"`
}

// Credits contains cast and crew.
type Credits struct {
	Cast []CastMember `json:"cast"`
	Crew []CrewMember `json:"crew"`
}

// CastMember is an actor in a movie.
type CastMember struct {
	Name      string `json:"name"`
	Character string `json:"character"`
	Order     int    `json:"order"`
}

// CrewMember is a crew member (director, writer, etc.).
type CrewMember struct {
	Name string `json:"name"`
	Job  string `json:"job"`
}

// Reviews contains user reviews.
type Reviews struct {
	Results []Review `json:"results"`
}

// Review is a single user review.
type Review struct {
	Author  string `json:"author"`
	Content string `json:"content"`
}

// Keywords contains movie keywords.
type Keywords struct {
	Keywords []Keyword `json:"keywords"`
}

// Keyword is a single keyword tag.
type Keyword struct {
	Name string `json:"name"`
}
```

package tmdb

import (
	"context"
	"encoding/json"
	"fmt"
	"net/http"
	"strconv"
	"strings"
	"sync"
)

// targetGenres are the 5 genre names we fetch for the demo KB.
var targetGenres = []string{"Action", "Comedy", "Drama", "Science Fiction", "Animation"}

// perGenreCount maps size to number of movies per genre.
var perGenreCount = map[string]int{
	"small": 10,
}

// Client is a TMDB API client.
type Client struct {
	baseURL    string
	apiKey     string
	httpClient *http.Client
}

// NewClient creates a TMDB client. Pass nil for httpClient to use http.DefaultClient.
func NewClient(baseURL, apiKey string, httpClient *http.Client) *Client {
	if httpClient == nil {
		httpClient = http.DefaultClient
	}
	return &Client{baseURL: strings.TrimRight(baseURL, "/"), apiKey: apiKey, httpClient: httpClient}
}

// FetchTopByGenres fetches top-rated movies across target genres.
// Only "small" size is supported; others return an error.
func (c *Client) FetchTopByGenres(ctx context.Context, size string) ([]MovieDetail, error) {
	count, ok := perGenreCount[size]
	if !ok {
		return nil, fmt.Errorf("unsupported size %q: only \"small\" is implemented", size)
	}

	genres, err := c.fetchGenres(ctx)
	if err != nil {
		return nil, fmt.Errorf("fetch genres: %w", err)
	}

	// Map genre names to IDs.
	genreIDs := make(map[string]int, len(genres))
	for _, g := range genres {
		genreIDs[g.Name] = g.ID
	}

	// Discover top-rated movies per genre, deduplicate by ID.
	seen := make(map[int]struct{})
	var movieIDs []int

	for _, name := range targetGenres {
		gid, ok := genreIDs[name]
		if !ok {
			continue
		}
		summaries, err := c.discoverMovies(ctx, gid, count)
		if err != nil {
			return nil, fmt.Errorf("discover genre %s: %w", name, err)
		}
		for _, s := range summaries {
			if _, dup := seen[s.ID]; !dup {
				seen[s.ID] = struct{}{}
				movieIDs = append(movieIDs, s.ID)
			}
		}
	}

	// Fetch full details concurrently (max 10 goroutines).
	type result struct {
		movie MovieDetail
		err   error
	}
	results := make([]result, len(movieIDs))
	var wg sync.WaitGroup
	sem := make(chan struct{}, 10)

	for i, id := range movieIDs {
		wg.Add(1)
		go func(idx, movieID int) {
			defer wg.Done()
			sem <- struct{}{}
			defer func() { <-sem }()
			m, err := c.fetchMovieDetail(ctx, movieID)
			results[idx] = result{movie: m, err: err}
		}(i, id)
	}
	wg.Wait()

	var movies []MovieDetail
	for _, r := range results {
		if r.err != nil {
			return nil, fmt.Errorf("fetch movie detail: %w", r.err)
		}
		movies = append(movies, r.movie)
	}
	return movies, nil
}

func (c *Client) get(ctx context.Context, path string) (*http.Response, error) {
	url := c.baseURL + path
	if strings.Contains(url, "?") {
		url += "&api_key=" + c.apiKey
	} else {
		url += "?api_key=" + c.apiKey
	}
	req, err := http.NewRequestWithContext(ctx, http.MethodGet, url, nil)
	if err != nil {
		return nil, err
	}
	return c.httpClient.Do(req)
}

func (c *Client) fetchGenres(ctx context.Context) ([]Genre, error) {
	resp, err := c.get(ctx, "/genre/movie/list")
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()
	var result GenreListResponse
	return result.Genres, json.NewDecoder(resp.Body).Decode(&result)
}

func (c *Client) discoverMovies(ctx context.Context, genreID, count int) ([]MovieSummary, error) {
	path := fmt.Sprintf("/discover/movie?sort_by=vote_average.desc&with_genres=%d&vote_count.gte=1000&page=1", genreID)
	resp, err := c.get(ctx, path)
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()
	var result DiscoverResponse
	if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
		return nil, err
	}
	if len(result.Results) > count {
		result.Results = result.Results[:count]
	}
	return result.Results, nil
}

func (c *Client) fetchMovieDetail(ctx context.Context, movieID int) (MovieDetail, error) {
	path := "/movie/" + strconv.Itoa(movieID) + "?append_to_response=credits,reviews,keywords"
	resp, err := c.get(ctx, path)
	if err != nil {
		return MovieDetail{}, err
	}
	defer resp.Body.Close()
	var detail MovieDetail
	return detail, json.NewDecoder(resp.Body).Decode(&detail)
}

package tmdb

import (
	"fmt"
	"strings"
)

// RenderMarkdown converts a MovieDetail into a markdown document suitable for RAG chunking.
func RenderMarkdown(m MovieDetail) string {
	var b strings.Builder

	// Title with year
	year := ""
	if len(m.ReleaseDate) >= 4 {
		year = m.ReleaseDate[:4]
	}
	fmt.Fprintf(&b, "# %s (%s)\n\n", m.Title, year)

	// Overview
	if m.Overview != "" {
		b.WriteString("## Overview\n\n")
		b.WriteString(m.Overview)
		b.WriteString("\n\n")
	}

	// Details
	b.WriteString("## Details\n\n")

	if len(m.Genres) > 0 {
		names := make([]string, len(m.Genres))
		for i, g := range m.Genres {
			names[i] = g.Name
		}
		fmt.Fprintf(&b, "- **Genres:** %s\n", strings.Join(names, ", "))
	}

	// Director from crew
	for _, c := range m.Credits.Crew {
		if c.Job == "Director" {
			fmt.Fprintf(&b, "- **Director:** %s\n", c.Name)
			break
		}
	}

	// Top 5 cast
	if len(m.Credits.Cast) > 0 {
		limit := 5
		if len(m.Credits.Cast) < limit {
			limit = len(m.Credits.Cast)
		}
		names := make([]string, limit)
		for i := 0; i < limit; i++ {
			names[i] = m.Credits.Cast[i].Name
		}
		fmt.Fprintf(&b, "- **Cast:** %s\n", strings.Join(names, ", "))
	}

	if m.VoteAverage > 0 {
		fmt.Fprintf(&b, "- **Rating:** %.1f/10 (%d votes)\n", m.VoteAverage, m.VoteCount)
	}
	if m.Runtime > 0 {
		fmt.Fprintf(&b, "- **Runtime:** %d min\n", m.Runtime)
	}

	if len(m.Keywords.Keywords) > 0 {
		names := make([]string, len(m.Keywords.Keywords))
		for i, k := range m.Keywords.Keywords {
			names[i] = k.Name
		}
		fmt.Fprintf(&b, "- **Keywords:** %s\n", strings.Join(names, ", "))
	}

	// Reviews
	if len(m.Reviews.Results) > 0 {
		b.WriteString("\n## Reviews\n\n")
		limit := 3
		if len(m.Reviews.Results) < limit {
			limit = len(m.Reviews.Results)
		}
		for i := 0; i < limit; i++ {
			r := m.Reviews.Results[i]
			// Truncate long reviews to 500 chars
			content := r.Content
			if len(content) > 500 {
				content = content[:500] + "..."
			}
			fmt.Fprintf(&b, "> %q — %s\n\n", content, r.Author)
		}
	}

	return b.String()
}
```

// SeedService orchestrates demo org/KB creation and TMDB document ingestion.
type SeedService struct {
	orgSvc  OrgServiceI
	wsSvc   WorkspaceServiceI
	kbSvc   KBServiceI
	docSvc  DocumentServiceI
	tmdb    *tmdb.Client
	storage storage.Client

// NewSeedService creates a new SeedService.
func NewSeedService(
	orgSvc OrgServiceI,
	wsSvc WorkspaceServiceI,
	kbSvc KBServiceI,
	docSvc DocumentServiceI,
	tmdbClient *tmdb.Client,
	storageClient storage.Client,
	queueClient *queue.Client,
) *SeedService {
	return &SeedService{
		orgSvc:  orgSvc,
		wsSvc:   wsSvc,
		kbSvc:   kbSvc,
		docSvc:  docSvc,
		tmdb:    tmdbClient,
		storage: storageClient,
		queue:   queueClient,
	}
}

// SeedDemo creates the demo org, workspace, KB, fetches TMDB data, and enqueues document processing.
// Idempotent: if the demo org already exists, returns current state without re-seeding.
func (s *SeedService) SeedDemo(ctx context.Context, size string) (*handler.SeedResult, error) {
	// Step 1: Check idempotency — look up demo org by slug.
	existingOrg, err := s.orgSvc.GetBySlug(ctx, "demo")
	if err == nil && existingOrg != nil {
		// Demo org exists — find the KB and return current state.
		return s.buildExistingResult(ctx, existingOrg)
	}

	// Step 2: Create demo org.
	org, err := s.orgSvc.Create(ctx, model.CreateOrgRequest{Name: "Raven Demo"})
	if err != nil {
		return nil, fmt.Errorf("create demo org: %w", err)
	}

	// Step 3: Create workspace.
	ws, err := s.wsSvc.Create(ctx, org.ID, model.CreateWorkspaceRequest{Name: "Movies"})
	if err != nil {
		return nil, fmt.Errorf("create demo workspace: %w", err)
	}

	// Step 4: Create knowledge base.
	kb, err := s.kbSvc.Create(ctx, org.ID, ws.ID, model.CreateKBRequest{
		Name:        "Movie Database",
		Description: "Top-rated films from TMDB — explore with semantic search.",
	})
	if err != nil {
		return nil, fmt.Errorf("create demo KB: %w", err)
	}

	// Step 5: Fetch TMDB data.
	movies, err := s.tmdb.FetchTopByGenres(ctx, size)
	if err != nil {
		return nil, fmt.Errorf("fetch TMDB movies: %w", err)
	}

	// Step 6: Create documents and enqueue processing.
	enqueued := 0
	for _, movie := range movies {
		md := tmdb.RenderMarkdown(movie)
		filename := fmt.Sprintf("%s.md", toSlug(movie.Title))

		// Upload to SeaweedFS.
		fid, err := s.storage.Upload(ctx, filename, strings.NewReader(md))
		if err != nil {
			slog.WarnContext(ctx, "seed: failed to upload movie document", "movie", movie.Title, "error", err)
			continue
		}

		// Create document record.
		doc, err := s.docSvc.CreateFromSeed(ctx, org.ID, kb.ID, filename, "text/markdown", fid, int64(len(md)))
		if err != nil {
			slog.WarnContext(ctx, "seed: failed to create document record", "movie", movie.Title, "error", err)
			continue
		}

		// Enqueue processing job.
		if err := s.queue.EnqueueDocumentProcess(ctx, queue.DocumentProcessPayload{
			DocumentID:     doc.ID,
			OrgID:          org.ID,
			KnowledgeBaseID: kb.ID,
		}); err != nil {
			slog.WarnContext(ctx, "seed: failed to enqueue document", "doc_id", doc.ID, "error", err)
			continue
		}
		enqueued++
	}

	slog.InfoContext(ctx, "seed: demo KB seeded", "org_id", org.ID, "kb_id", kb.ID, "enqueued", enqueued)

	return &handler.SeedResult{
		OrgID:             org.ID,
		WorkspaceID:       ws.ID,
		KBID:              kb.ID,
		DocumentsEnqueued: enqueued,
		PipelineStatus:    "seeding",
	}, nil
}

func (s *SeedService) buildExistingResult(ctx context.Context, org *model.Organization) (*handler.SeedResult, error) {
	// Return a result with existing IDs — pipeline status based on doc processing state.
	return &handler.SeedResult{
		OrgID:          org.ID,
		PipelineStatus: "ready",
	}, nil
}
